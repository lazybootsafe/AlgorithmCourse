int check(char date[]) //判断日期是否合法//
{
	int year, month, day, i, leap; //要求格式判断//
	for (i = 0, year = 0; i<4; i++)
		year = year * 10 + (date[i] - '0');
	for (i = 5, month = 0; i<7; i++)
		month = month * 10 + (date[i] - '0');
	for (i = 8, day = 0; i<10; i++)
		day = day * 10 + (date[i] - '0');
	if (month<0 || month>12)
		return 0;
	else //日期存在判断//
	{
		leap = (year % 4 == 0 && year % 100 != 0 || year % 400 == 0);
		if (month == 1 || month == 3 || month == 5 || month == 7 || month == 8 || month == 10 || month == 12)
		{
			if (day<0 || day>31)
				return 0;
			else
				return 1;
		}
		if (month == 4 || month == 6 || month == 9 || month == 11)
		{
			if (day<0 || day>30)
				return 0;
			else
				return 1;
		}
		if (leap == 1 && month == 2)
		{
			if (day<0 || day>29)
				return 0;
			else
				return 1;
		}
		if (leap == 0 && month == 2)
		{
			if (day<0 || day>28)
				return 0;
			else
				return 1;
		}
	}
	return 0;
}

/*从磁盘文件导入质检信息到内存*/
void loadRec()
{
	FILE *fp;
	int n = 0;
	if ((fp = fopen("LY.txt", "r")) == NULL) /*打开文件*/
	{
		printf("文件不存在或无法打开！ \n");
		system("pause");
		return;
	}
	head = p1 = (struct autoparts *)malloc(LEN); /*开辟一个新单元*/
	memset(p1, 0, LEN);
	p1->next = NULL;
	while (!feof(fp))
	{
		n++;
		fscanf(fp, "%s%s%s%s%s%s%s%s\n", p1->qid, p1->qname, p1->producer, p1->pd,
			p1->qresult, p1->qproblem, p1->qpersons, p1->time);
		p2 = (struct autoparts *)malloc(LEN);
		p1->next = p2;
		p3 = p1;
		p1 = p2;
		p1->next = NULL;
	}
	free(p2);
	p3->next = NULL;
	printf("\n %d 项信息输入完毕， 按任意键返回......\n", n);
	getch();
	system("cls"); /*清屏*/
	writeinfo_flag = 0; /*设置保存记录标志*/
	if (fclose(fp)) /* 关闭文件 */
	{
			printf("Can not close the file!\n");
		exit(0);
	}
	return;
}

	{
		p2 = p1;
		p1 = p1->next;

	{
		if (head->next == NULL)
		{
			free(head);
			head = NULL;
		}
		else if (head == p1)
		{
			head = p1->next;

		}

			p2->next = p1->next;
			free(p1);
		}
		printf("\n 删除完毕； ");

		printf("\n\n 找不到该质检号信息!");
