# 求一维数组的最大值及其下标


```cpp
#include <stdio.h>

/*
本题要求编写程序，找出给定的n个数中的
最大值及其对应的最小下标（下标从0开始）。
*/

int main(int argc, char *argv[]) {
 int n,m,i,x=1;                                          
 //printf("请输入一个整数：");
 scanf("%d",&n);
 int a[10];
 //printf("请输入%d个数字",n);
 for(i=0;i<n;i++){
  scanf("%d",&a[i]);
 }
 
 m=a[0];
 
 for(i=0;i<n;i++){
  while(a[i]>m){
   m=a[i];                                                  //计算最大值
   x++;                                                      //x计算最小下标
  }
 }
 printf("%d %d",m,x);
 return 0;
}

```

# 将数组中的数逆序存放

```cpp
#include<stdio.h>

		scanf("%d",&x);
		str[i]=x;
	}
	for(i=0;i<n/2;i++)
	{
		temp=str[i];

	}

	{
		printf("%d",str[i]);
		if(i!=n-1)
		{
			printf(" ");//保证最后一个数后没空格
		}
	}
	return 0;
 }
```

# 找出不是两个数组共有的元素

```cpp
#include <stdio.h>

int delx(int d[] , int n , int e)
{
        int i , j , k;
        for(k = n , i = 0 ; i < k ;) {
                if(d[i] == e) {
                        for(j = i + 1 ; j < k ; j ++) d[j - 1] = d[j];
                        k --;
                } else {
                        i ++;
                }
        }
        return k;
}

int main(void)
{
        int a , b , c , d[2][500] , i , j , k , m , n;
        bool f;
        scanf("%d" , & n);
        for(i = 0 ; i < n ; i ++) scanf("%d" , & d[0][i]);
        scanf("%d" , & m);
        for(i = 0 ; i < m ; i ++) scanf("%d" , & d[1][i]);
        for(b = n , c = m , i = 0 ; i < b ;) {
                for(f = false , j = 0 ; j < c ; j ++) {
                        if(d[0][i] == d[1][j]) {
                                f = true;
                                break;
                        }
                }
                if(f) {
                        a = d[0][i];
                        b = delx(d[0] , b , a);
                        c = delx(d[1] , c , a);
                        i = 0;
                } else {
                        i ++;
                }
        }
        if(b) {
                printf("%d" , d[0][0]);
                for(k = 1 ; k < b ; k ++) printf(" %d" , d[0][k]);
        }
        if(c) {
                if(! b) printf("%d" , d[1][0]);
                else printf(" %d" , d[1][0]);
                for(k = 1 ; k < c ; k ++) printf(" %d" , d[1][k]);
        }
        printf("\n");
}
```

# 求一批整数中出现最多的数字

```cpp
#include <stdio.h>
int main()
{
    int i, n, num, max;
    int a[100] = {0};//存放出现的数
         //printf("输入几个正整数:");
    scanf("%d", &n);

    for (i=0; i<n; i++)
    {
        scanf("%d", &num);
        while (num != 0) 
        {
            a[num%10]++;
            num /= 10;
        }

    }

    max = a[0];//初始化最大次数为0
    for (i=1; i<100; i++) 
    {
        if (a[i] > max)
        {
            max = a[i];
        }
    }
    //printf("出现次数最多%d次的数字是:", max);
    //printf("%d", max);
    for (i=0; i<100; i++)
    {
        if (a[i] == max)
        {
            printf(" %d", i);//输出出现次数最多的个数
        }
    }
    printf("\n");

    return 0;
}



```


# 判断上三角矩阵

```cpp
#include <stdio.h>  
  //上三角矩阵指主对角线以下的元素都为0的矩阵；主对角线为从矩阵的左上角至右下角的连线。
//本题要求编写程序，判断一个给定的方阵是否上三角矩阵。
int main() {  
//输入第一行给出一个正整数T，为待测矩阵的个数。接下来给出T个矩阵的信息：
//每个矩阵信息的第一行给出一个不超过10的正整数n。随后n行，每行给出n个整数，其间以空格分隔。
	int t,n,i,j,k;
	int a[10][10]; 
	scanf("%d",&t);
	int count=0;
	int b[100]={0};  //记录判断记录 
	
	for(k=0; k<t; k++){
		//读入矩阵 
		scanf("%d",&n);
		for(i=0; i<n; i++){
			for(j=0; j<n; j++){
				scanf("%d",&a[i][j]);	
			}		
		}
		//判断矩阵 
		for(i=0; i<n; i++){
			for(j=0; j<n; j++){
				if(i>j && a[i][j] != 0){
					count++;
				}
			}		
		}
		if(count==0 ){
    		b[k] = 1;
    	}else{
    		b[k] = 0;
    	}
    	count=0;		
	}
	
	for(k=0; k<t; k++){
		if(b[k]==1){
			printf("YES\n");
		}else{
			printf("NO\n");
		} 
	}
	
    return 0;  
}
```
