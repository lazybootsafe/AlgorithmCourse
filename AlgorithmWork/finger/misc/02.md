```cpp
#include <stdio.h>
#include <math.h>
int fn(int a, int n);
int main()
{
	int a = 0, n = 0, sum = 0;
	scanf("%d", &a);//scanf()里不能有换行符 
	scanf("%d", &n);
	for (int j = n; j>0; j--)
	{
		sum = sum + fn(a, j);
	}
	printf("Sum=%d", sum);
	return 0;
}

int fn(int a, int n)
{
	int result = 0;
	for (int i = 0; i<n; i++)
	{
		result = result + a*pow(10, i);//pow(2,3)表示2的3次 	
	}
	return result;
}
```

```cpp
//使用函数累加由n个a构成的整数之和
#include<stdio.h> 
int fn(int a,int n);
int main(int argc,char const *argv[])
{
	int a,n;
	printf("请输入两个正整数a和n:");
	while(1)
	{
		scanf("%d%d",&a,&n);
		if(a>0&&a<=9&&n>0)
			break;
		printf("请重新输入:");
	}
	
	printf("fn(%d,%d)=%d\n",a,n,fn(a,n));
	
	return 0;
}
 
int fn(int a,int n)
{
	int sum=0;
	int origin=a;
	while(n)
	{
		sum+=a;
		a=a*10+origin;
		n--;
	}
	
	return sum;
}
```
