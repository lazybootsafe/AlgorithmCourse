## 使用函数累加由n个a构成的整数之和


正确求和 应该是 3+33=36 才对
```cpp
#include <stdio.h>
#include <math.h>
int fn(int a, int n);
int main()
{
	int a = 0, n = 0, sum = 0;
	scanf("%d", &a);//scanf()里不能有换行符 
	scanf("%d", &n);
	for (int j = n; j>0; j--)
	{
		sum = sum + fn(a, j);
	}
	printf("Sum=%d", sum);
	return 0;
}

int fn(int a, int n)
{
	int result = 0;
	for (int i = 0; i<n; i++)
	{
		result = result + a*pow(10, i);//pow(2,3)表示2的3次 	
	}
	return result;
}
```
